/**
 * 时间转化函数，时间戳入参为秒
*/
convertTime: function (sec) {
  let time
  if (sec > -1) {
    let hour = Math.floor(sec / 3600)
    let min = Math.floor(sec / 60) % 60
    let second = sec % 60
    let day = parseInt(hour / 24)
    hour = hour - 24 * day
    hour = hour < 10 ? `0${hour}` : hour
    time = day < 10 ? `0${day}:${hour}:`: `${day}:${hour}:`
    if (min < 10) {
      time += "0"
    }
    time += min + ":";
    if (second < 10) {
      time += "0"
    }
    time += second
    return time.split(':') // 返回一个包含天时分秒的数组
  }
}
